```sql
            WHEN ABS(bt.osszeg - s.vegosszeg) / s.vegosszeg < 0.10 THEN 20
            ELSE 0
        END AS osszeg_score,
        -- KÃ¶zlemÃ©ny pontszÃ¡m (35)
        CASE WHEN bt.kozlemeny LIKE '%' || s.szamlaszam || '%' THEN 35 ELSE 0 END AS kozlemeny_score,
        -- Partner pontszÃ¡m (15)
        CASE WHEN p.nev LIKE '%' || bt.partner_nev || '%' THEN 15 ELSE 0 END AS partner_score,
        -- DÃ¡tum pontszÃ¡m (10)
        CASE
            WHEN ABS(bt.datum - s.kelt) <= 3 THEN 10
            WHEN ABS(bt.datum - s.kelt) <= 7 THEN 5
            ELSE 0
        END AS datum_score
    FROM BANK_TRANZAKCIO bt
    CROSS JOIN SZAMLA s
    LEFT JOIN PARTNER p ON s.partner_id = p.partner_id
    WHERE bt.parositas_statusz = 'FELDOLGOZATLAN'
      AND s.fizetve = FALSE
)
SELECT
    tranzakcio_id,
    szamla_id,
    (osszeg_score + kozlemeny_score + partner_score + datum_score) AS total_score
FROM score_calc
WHERE (osszeg_score + kozlemeny_score + partner_score + datum_score) >= 50
ORDER BY total_score DESC;
```
